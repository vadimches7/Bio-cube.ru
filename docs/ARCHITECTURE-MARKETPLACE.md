```sql
-- Organizations (Tenants)
CREATE TABLE organizations (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name TEXT NOT NULL,
  slug TEXT UNIQUE NOT NULL,
  domain TEXT,
  logo_url TEXT,
  theme_colors JSONB,
  settings JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Users
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  email TEXT UNIQUE NOT NULL,
  organization_id UUID REFERENCES organizations(id),
  role TEXT CHECK (role IN ('owner', 'admin', 'user')),
  profile_data JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Pages
CREATE TABLE pages (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  organization_id UUID REFERENCES organizations(id),
  slug TEXT NOT NULL,
  title TEXT NOT NULL,
  blocks JSONB NOT NULL, -- Array of block configs
  seo_settings JSONB,
  published BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(organization_id, slug)
);

-- Forms
CREATE TABLE forms (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  organization_id UUID REFERENCES organizations(id),
  name TEXT NOT NULL,
  fields JSONB NOT NULL, -- Form field definitions
  webhook_url TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Leads
CREATE TABLE leads (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  organization_id UUID REFERENCES organizations(id),
  form_id UUID REFERENCES forms(id),
  data JSONB NOT NULL, -- Form submission data
  utm_params JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Media
CREATE TABLE media (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  organization_id UUID REFERENCES organizations(id),
  url TEXT NOT NULL,
  type TEXT, -- 'image', 'video', 'document'
  metadata JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```

-- Users can only see their organization's data
CREATE POLICY "Users see own org data" ON pages
